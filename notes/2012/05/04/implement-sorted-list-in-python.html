class SortedList(list):
    K = -1

    def __init__(self, K=-1):
        list.__init__(self)
        if K != -1: self.K = K

    def append(self, x):
        bisect.insort(self, x)
        if self.K != -1:
            if len(self)==self.K+1:
                self.pop(0)
