def foo1(*args):
    for arg in args: print arg

def foo2(**kargs):
    for key in kargs:
        print key, kargs[key]

def foo3(*args, **kargs):
    print "args:"
    for arg in args:
        print arg

    print "kargs":
    for key in kargs:
        print key, kargs[key]
